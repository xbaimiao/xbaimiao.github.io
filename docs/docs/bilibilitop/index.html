介绍

简介​

可以让你玩家绑定他自己的b站uid（无需登录只需填写uid） 然后他三连后可领取你配置的奖励

支持版本. 1.7.10-1.20.1

命令​

配置文件(GUI)​

title: '&c哔哩哔哩点赞系统'
sort:
  - 'XXXXXXXXX'
  - 'XA      X'
  - 'XXXXXXXXX'
items:
  X:
    material: 'STAINED_GLASS_PANE'
    name: '&c'
    lore:
      - '&c'
    customModelData: 10001
  A:
    material: 'ITEM_FRAME'
    name: '&a视频号: %bvid%'
    lore:
      - '&f[&e标题&f]'
      - '&f&l%title%'
      - '&f[&e详情&f]'
      - '&b点赞数量: &a%like%'
      - '&b投币数量: &a%coin%'
      - '&b收藏数量: &a%favorite%'
      - '&f'
      - '&f&l[&6&l操作&f&l]'
      - '&e&lQ &b键获取视频链接'
      - '&e&l单击 &b获取三连奖励'
      - '&f'
      - '&f&l[&6&l奖励&f&l]'
      - '&a&lVIP &f7&l天'
      - '&f500 있'
      - '&x&f&d&d&d&4&b&l贡献之心'
      - '&f1000&a&l经验'
    customModelData: 10002
    bVid: 'BV1ud4y1C7vC'
    # 奖励命令
    reward:
      - 'points give %player% 500'
      - 'mi give MATERIAL 贡献之心 %player%'
      - 'experience add %player% 1000'

配置文件(CONFIG)​

debug: false
proxy:
  # 是否启用代理
  enable: false
  # 代理类型 可选 HTTP SOCKS
  type: HTTP
  # 代理地址
  address: localhost
  # 代理端口
  port: 7890
mysql:
  enable: false
  host: 'localhost'
  port: 3306
  database: 'database'
  user: 'root'
  passwd: 'passwd'
  ssl: false
